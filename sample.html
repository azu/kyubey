key bind

Prev(←), Next(→), First(↑), Last(↓)

function bindKey() {
  $(window).keyup(function(e){
    if(e.keyCode==37) {         // ←
      goPrev();
    } else if(e.keyCode==38) {  // ↑
      goFirst();
    } else if(e.keyCode==39) {  // →
      goNext();
    } else if(e.keyCode==40) {  // ↓
      goLast();
    }
  });
};

page number

look at lower right

function paging() {
  var pageNum = $("section").length;
  var i = 1;
  $("section").each(function(){
    if ($(this).is(".current")) {
      return false; // break
    } else {
      i++;
    }
  });
  $("#page").text(i + " / " + pageNum);
};

animation

thanks to jQuery animation

function show(jObj) {
  $("section:visible").removeClass("current").hide();
  if (config.animation == "fadeIn") {
    jObj.fadeIn();
  } else if (config.animation == "slideDown") {
    jObj.slideDown();
  }
  jObj.css("display", "block")
      .addClass("current");
  paging();
};

def fizzbuzz(n)
  (1..n).each do |i|
    s = ""
    s += "fizz" if i % 3 == 0
    s += "buzz" if i % 5 == 0
    s = i if s == ""
    puts s
  end
end
